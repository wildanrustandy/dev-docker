# Fungsi untuk mengakses semua perintah dari dev-docker dari mana saja

# Fungsi untuk switching environment (hanya dari dalam directory)
dev_env() {
    cd /Users/wildanrustandy/Dev/dev-docker && ./switch_env.sh "$1"
}

# Fungsi untuk manajemen database (bisa dari mana saja)
dev_db() {
    /Users/wildanrustandy/Dev/dev-docker/db_manager.sh "$@"
}

# Fungsi untuk manajemen database (menggunakan make)
# Catatan: fungsi ini harus dijalankan dari dalam directory dev-docker
dev_db_make() {
    cd /Users/wildanrustandy/Dev/dev-docker && make "db-$1" DBNAME="$2"
}

# Fungsi untuk manajemen database dengan restore (menggunakan make)
# Catatan: fungsi ini harus dijalankan dari dalam directory dev-docker
dev_db_restore_make() {
    cd /Users/wildanrustandy/Dev/dev-docker && make db-restore DBNAME="$1" FILE="$2"
}

# Contoh penggunaan dari mana saja:
#   dev_env odoo14              # Switch to odoo14 environment (hanya jika dalam directory)
#   dev_env equip               # Switch to equip environment (hanya jika dalam directory)
#   dev_env equip_prj           # Switch to equip_prj environment (hanya jika dalam directory)
#
#   dev_db create 'MY_DATABASE' # Create database from anywhere
#   dev_db drop 'MY_DATABASE'   # Drop database from anywhere
#   dev_db list                 # List databases from anywhere
#   dev_db backup 'DB' 'file.sql'   # Backup database from anywhere
#   dev_db restore 'DB' 'file.sql'  # Restore database from anywhere
#
# Untuk membuat fungsi ini permanen, tambahkan ke ~/.zshrc atau ~/.bashrc:
#   cat >> ~/.zshrc << 'EOF'
#   # Dev-docker functions
#   dev_db() { /Users/wildanrustandy/Dev/dev-docker/db_manager.sh "$@"; }
#   dev_env() { cd /Users/wildanrustandy/Dev/dev-docker && ./switch_env.sh "$1"; }
#   EOF
#   source ~/.zshrc
